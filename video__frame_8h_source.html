54  MappedMat(GstBuffer *buffer, const GstVideoInfo *video_info, GstMapFlags flag = GST_MAP_READ) : buffer(buffer) {
55  if (!gst_buffer_map(buffer, &map_info, flag))
56  throw std::runtime_error("GVA::MappedMat: Could not map buffer to system memory");
57 
58  GstVideoFormat format = video_info->finfo->format;
59  int stride = video_info->stride[0];
60 
61  switch (format) {
62  case GST_VIDEO_FORMAT_BGR:
63  this->mat() = cv::Mat(cv::Size(video_info->width, video_info->height), CV_8UC3,
64  reinterpret_cast<char *>(map_info.data), stride);
65  break;
66  case GST_VIDEO_FORMAT_NV12:
67  this->mat() = cv::Mat(cv::Size(video_info->width, static_cast<int32_t>(video_info->height * 1.5f)), CV_8UC1,
68  reinterpret_cast<char *>(map_info.data), stride);
69  break;
70  case GST_VIDEO_FORMAT_BGRA:
71  case GST_VIDEO_FORMAT_BGRx:
72  this->mat() = cv::Mat(cv::Size(video_info->width, video_info->height), CV_8UC4,
73  reinterpret_cast<char *>(map_info.data), stride);
74  break;
75 
76  default:
77  throw std::runtime_error("GVA::MappedMat: Unsupported format");
78  }
79  }
80 
81  ~MappedMat() {
82  if (buffer != nullptr)
83  gst_buffer_unmap(buffer, &map_info);
84  }

91  return cv_mat;
92  }

271  RegionOfInterest add_region(int x, int y, int w, int h, int label_id, double confidence = 0.0,
272  GstStructure *region_tensor = nullptr) {
273  if (!this->is_bounded(x, y, w, h)) {
274  int x_init = x, y_init = y, w_init = w, h_init = h;
275  clip(x, y, w, h);
276  GST_DEBUG("ROI coordinates {x, y, w, h} are out of image borders and will be clipped: [%d, %d, %d, %d] "
277  "-> [%d, %d, %d, %d]",
278  x_init, y_init, w_init, h_init, x, y, w, h);
279  }
280 
281  const gchar *label = "";
282  get_label_by_label_id(region_tensor, label_id, &label);
283 
284  GstVideoRegionOfInterestMeta *meta = gst_buffer_add_video_region_of_interest_meta(buffer, label, x, y, w, h);
285 
286  if (not region_tensor)
287  region_tensor = gst_structure_new_empty("detection");
288  else
289  gst_structure_set_name(region_tensor, "detection"); // make sure we're about to add detection Tensor
290 
291  gst_structure_set(region_tensor, "label_id", G_TYPE_INT, label_id, "confidence", G_TYPE_DOUBLE, confidence,
292  "x_min", G_TYPE_DOUBLE, (double)x / info->width, "x_max", G_TYPE_DOUBLE,
293  (double)(x + w) / info->width, "y_min", G_TYPE_DOUBLE, (double)y / info->height, "y_max",
294  G_TYPE_DOUBLE, (double)(y + h) / info->height, NULL);
295 
296  _regions.emplace_back(meta);
297  _regions.back().add_tensor(region_tensor);
298  // region_tensor will be freed along with GstVideoRegionOfInterestMeta
299  return _regions.back();
300  }

410  void clip(int &x, int &y, int &w, int &h) {
411  x = (x < 0) ? 0 : (x > info->width) ? info->width : x;
412  y = (y < 0) ? 0 : (y > info->height) ? info->height : y;
413  w = (w < 0) ? 0 : (x + w > info->width) ? info->width - x : w;
414  h = (h < 0) ? 0 : (y + h > info->height) ? info->height - y : h;
415  }
